## Behavioral modeling

```verilog
module SR_behavioral(
	q,
	q_,
	en,
	s,
	r
	);

	output reg q;
	output q_;
	input en;
	input s;
	input r;

	assign q_ = ~q;

	always@(*)
	begin
		if(en)
		begin
			casex( {s, r} )
				2'b00: q <= q;
				2'b01: q <= 1'b0;
				2'b10: q <= 1'b1;
				2'b11: q <= 1'bx;			
			endcase
		end
	end
endmodule
```

module Tb_SR_LATCH();
	reg EN, S, R;
	wire Q, Q_;

	SR_behavioral SR( .q(Q), .q_(Q_), .en(EN), .s(S), .r(R) );

		S=0;

	end

endmodule
