# Design Source

```verilog
module JOHNSON #(parameter size = 4) (
    output [size-1:0]Q,
    input clk, clr_n
    );
    
    wire [size-1:0]Q_;
    
    genvar i;
    generate
        for(i=0; i<size; i=i+1)
        begin : ff
            if(i == 0)
                D_FF DFF(.Q(Q[i]), .Q_(Q_[i]), .clk(clk), .clr_n(clr_n), .D(Q_[size-1]));
            else
                D_FF DFF(.Q(Q[i]), .Q_(Q_[i]), .clk(clk), .clr_n(clr_n), .D(Q_[i-1]));
        end
    endgenerate
endmodule
```

module Tb_JOHNSON();

        clr_n=1'b0;
        
        clk=1'b0;
        forever

    end
