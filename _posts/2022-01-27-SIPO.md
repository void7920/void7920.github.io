# Design Source

```verilog
module SIPO #(parameter size=4) (
    o,
    clk,
    clear_n,
    i
    );
    
    output reg [size-1:0]o;
    input clk;
    input clear_n;
    input i;
    
    always@(posedge clk or negedge clear_n)
    begin
        if(!clear_n)
            o <= 0;
            
        else
            o <= {o[size-2:0], i};
    end
endmodule
```
