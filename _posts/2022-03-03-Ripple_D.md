# Design Source

```verilog
module Ripple_D #(parameter size = 4)(
  output [size-1:0]Q,
  input clk, clr
  );
    
  wire [size-1:0]Q_;
 
  genvar i;
  generate
    for(i=0; i<size; i=i+1)
      begin : Neg_DFF
        if(i == 0)
          D_FF_neg neg_D(.Q(Q[i]), .Q_(Q_[i]), .clk(clk), .clr(clr), .D(Q_[i]));
        else
          D_FF_neg neg_D(.Q(Q[i]), .Q_(Q_[i]), .clk(Q[i-1]), .clr(clr), .D(Q_[i]));
        end
    endgenerate  
endmodule
```
